def add_extension_if_missing(filename, default_extension="py"):
    """Check if filename has an extension and add the default if missing.

    If the provided filename has no extension, appends the default extension.
    Otherwise, returns the filename unchanged.

    Args:
        filename (str): The filename to check.
        default_extension (str): The extension to add if missing. Defaults to 'py'.

    Returns:
        str: The filename with extension added if necessary.

    Example:
        >>> add_extension_if_missing('script')
        'script.py'
        >>> add_extension_if_missing('script.txt')
        'script.txt'
    """
    path = Path(filename)
    if path.suffix == "":  # No extension found
        return str(path.with_suffix(f".{default_extension}"))
    return filename

def printc(text, color="white", sep=" ", end="\n", file=sys.stdout, flush=False):
    """Print colored text to console using ANSI escape codes.

    Supports various colors and automatically logs to Robot Framework if called
    from within a Robot Framework context.

    Args:
        text (str): The text to print.
        color (str): Color name. Options: black, red, green, yellow, blue, magenta,
            cyan, white, error, debug, info, warning, critical, success.
            Defaults to 'white'.
        sep (str): Separator between arguments. Defaults to ' '.
        end (str): String appended after the last value. Defaults to '\n'.
        file: File object to write to. Defaults to sys.stdout.
        flush (bool): Whether to forcibly flush the stream. Defaults to False.

    Note:
        When called from Robot Framework, output is also logged to the framework's console.
    """
    colors = {
        "black": "\033[30m",
        "red": "\033[31m",
        "green": "\033[32m",
        "yellow": "\033[33m",
        "blue": "\033[34m",
        "magenta": "\033[35m",
        "cyan": "\033[36m",
        "white": "\033[37m",
        "error": "\033[31m",  # red
        "debug": "\033[37m",  # white
        "info": "",  # default
        "warning": "\033[33m",  # yellow
        "critical": "\033[31m",  # red
        "success": "\033[32m",  # green
    }
    reset = "\033[0m"
    color_code = colors.get(color.lower(), colors["white"])
    s = f"{color_code}{text}{reset}"
    if is_called_from_robot():
        logger.console("\n" + s)  # Also print log to console
    else:
        print(f"{s}", sep=sep, end=end, file=file, flush=flush)

def set_log_level(level):
    """Set the global log level for message filtering.

    Args:
        level (str): Log level to set. Supported values: DEBUG, INFO, SUCCESS,
            WARNING, ERROR, CRITICAL. Case-insensitive.

    Raises:
        No exceptions raised; invalid levels are logged as info.

    Note:
        The log level is stored in the global LOG_LEVEL variable.
    """
    global LOG_LEVEL
    LOG_LEVEL = level.upper()
    log_message("INFO", f"Log level is set to [{LOG_LEVEL}]")

def get_log_level_priority(level_name):
    """Get the numeric priority of a log level.

    Used internally for log level filtering. Higher priority values indicate
    more critical messages (e.g., CRITICAL has higher priority than DEBUG).

    Args:
        level_name (str): The log level name. Case-insensitive.

    Returns:
        int: Priority value. DEBUG=1, INFO/SUCCESS=2, WARNING=3, ERROR=4,
            CRITICAL=5. Returns 0 for unknown levels.
    """
    priorities = {"DEBUG": 1, "INFO": 2, "SUCCESS": 2, "WARNING": 3, "ERROR": 4, "CRITICAL": 5}
    return priorities.get(level_name.upper(), 0)

def log_message(level, message, level_attr=None, show_datetime=True):
    """Log a message with level filtering and optional timestamp.

    Checks multiple sources for the minimum log level to display:
    global LOG_LEVEL variable, __main__ module, or Robot Framework variable.

    Args:
        level (str): Message level. Supported values: DEBUG, INFO, SUCCESS,
            WARNING, ERROR, CRITICAL. Case-insensitive.
        message (str): The message text to log.
        level_attr (str): Minimum level to display. If None, uses LOG_LEVEL
            from globals, __main__, or Robot Framework. Defaults to 'INFO'.
        show_datetime (bool): Whether to prepend timestamp to message.
            Defaults to True.

    Note:
        Output format: [YYYY-MM-DD HH:MM:SS.fff] [LEVEL]: message
    """
    # Check for LOG_LEVEL in global var
    level_attr_glob = globals().get("LOG_LEVEL", None)
    if level_attr_glob:
        level_attr = level_attr_glob

    # Check for LOG_LEVEL in __main__
    if level_attr is None:
        try:
            import __main__

            level_attr = getattr(__main__, "LOG_LEVEL", "INFO")
        except:
            level_attr = "INFO"

        if is_called_from_robot():
            builtin = BuiltIn()
            level_attr = builtin.get_variable_value("${LOG_LEVEL}", "INFO")

    dt = f"[{datetime.now().strftime('%Y-%m-%d %H:%M:%S.%f')[:-3]}] " if show_datetime else ""
    if get_log_level_priority(level) >= get_log_level_priority(level_attr):
        s = f"{dt}[{level.upper()}]: {message}"
        printc(s, color=level)

def ask_yes_no(question, default="N"):
    """Prompt user for a yes/no question interactively.

    Displays a question and waits for user input ('y' or 'n'). Repeats
    until valid input is received.

    Args:
        question (str): The question to display to the user.
        default (str): Default choice if user presses Enter without input.
            'y' for yes, 'n' or 'N' for no. Defaults to 'N'.

    Returns:
        bool: True if user selects 'yes', False if 'no'.

    Example:
        >>> response = ask_yes_no("Do you want to continue?", default="Y")
    """
    prompt = " [y,N]: " if default.upper() == "N" else " [Y,n]: "
    while True:
        choice = input(question + prompt).strip().lower()
        if choice == "" and default:
            return default.lower() == "y"
        if choice in ["y", "yes"]:
            return True
        if choice in ["n", "no"]:
            return False
        printc("Please respond with 'y' or 'n'.", color="yellow")

def ensure_folder(path, interactive=True, relative_to_caller=True):
    """Ensure a folder exists, optionally creating it with user confirmation.

    Checks if a folder exists at the specified path. If it doesn't exist,
    optionally prompts the user before creating it.

    Args:
        path (str): Directory path to ensure. Can be absolute or relative.
        interactive (bool): If True, ask user before creating the directory.
            Defaults to True.
        relative_to_caller (bool): If True, interpret path as relative to
            the caller's file directory. Defaults to True.

    Returns:
        bool: True if folder exists or was created successfully, False otherwise.

    Raises:
        No exceptions are raised; errors are caught and reported.

    Example:
        >>> ensure_folder('./output', interactive=True)
        True
    """
    if relative_to_caller:
        caller_dir = Path(sys.argv[0]).resolve().parent
        path = os.path.join(caller_dir, path)

    if os.path.exists(path):
        return True
    if interactive:
        if not ask_yes_no(f"Folder '{path}' does not exist. Create it?"):
            printc("Operation cancelled.", color="red")
            return False
    try:
        os.makedirs(path, exist_ok=True)
        printc(f"Directory created: {path}", color="green")
        return True
    except Exception as e:
        printc(f"Error creating directory: {e}", color="red")
        return False

def list_picker(items, title="Select an item"):
    """Display an interactive TUI list picker for user selection.

    Shows a numbered list of items and prompts the user to select one by
    entering its number.

    Args:
        items (list): List of strings to choose from.
        title (str): Title to display above the list. Defaults to 'Select an item'.

    Returns:
        list: A list [selected_item, index] where selected_item is the chosen string
            and index is its position (0-based). Returns None if list is empty
            or user cancels.

    Example:
        >>> result = list_picker(['Option A', 'Option B', 'Option C'])
        >>> if result:
        ...     item, idx = result
        ...     print(f"Selected: {item} at index {idx}")
    """
    if not items:
        printc("List is empty.", color="yellow")
        return None

    selected_index = 0

    printc(f"\n{title}", color="blue")
    for i, item in enumerate(items):
        print(f"    <{i+1}>    {item}")

    print("Press number to select, or 'q' to cancel.")
    choice = input().strip().lower()
    if choice == "q":
        printc("Selection cancelled.", color="yellow")
        return None
    return (
        [items[int(choice) - 1], int(choice) - 1]
        if choice.isdigit() and 1 <= int(choice) <= len(items)
        else None
    )

def is_called_from_robot():
    """Check if the current code is being executed from Robot Framework.

    Inspects the call stack to determine if any frame belongs to Robot Framework.

    Returns:
        bool: True if called from Robot Framework, False otherwise.

    Note:
        Useful for conditional logging and behavior in functions that can be
        called from both Robot Framework and standard Python code.
    """
    for frame_info in inspect.stack():
        module = inspect.getmodule(frame_info.frame)
        if module and module.__name__.startswith("robot."):
            return True
    return False

def is_valid_filename(name):
    """Validate if a string is a valid filename for most filesystems.

    Checks for forbidden characters, length limits, and reserved names.
    Compatible with Windows and Linux/Unix systems.

    Args:
        name (str): The filename to validate.

    Returns:
        bool: True if filename is valid, False otherwise.

    Note:
        - Maximum length: 255 characters
        - Forbidden characters: < > : " / \ | ? * and control characters
        - Reserved names (Windows): CON, PRN, AUX, NUL, COM1, LPT1

    Example:
        >>> is_valid_filename('document.txt')
        True
        >>> is_valid_filename('file<name>.txt')
        False
    """

    # Forbidden characters on Windows/Linux
    forbidden = r'[<>:"/\\|?*\x00-\x1f]'

    # Check length (max 255 chars on most filesystems)
    if len(name) == 0 or len(name) > 255:
        return False

    # Check for forbidden characters
    if re.search(forbidden, name):
        return False

    # Check for reserved names (Windows)
    reserved = {"CON", "PRN", "AUX", "NUL", "COM1", "LPT1"}
    if name.upper() in reserved:
        return False

    return True

def list_directory(path=".", content="folder", match_extension=None):
    """List subdirectories or files in a given path with optional filtering.

    Uses pathlib for cross-platform compatibility.

    Args:
        path (str or Path): Directory path to list. Defaults to current directory.
        content (str): Type of items to list. Options: 'folder', 'file', 'all'.
            Defaults to 'folder'.
        match_extension (str): Optional file extension filter (without the dot).
            Example: 'py' to match only .py files. Only applies to files.

    Returns:
        list: Sorted list of item names (strings). Returns False on error.

    Raises:
        FileNotFoundError: If path does not exist.
        NotADirectoryError: If path is not a directory.

    Example:
        >>> list_directory('.', content='file', match_extension='py')
        ['script1.py', 'script2.py']
    """
    try:
        p = Path(path)

        if not p.exists():
            raise FileNotFoundError(f"Path '{path}' does not exist.")
        if not p.is_dir():
            raise NotADirectoryError(f"Path '{path}' is not a directory.")

        if content.lower() == "folder":
            items = [f for f in p.iterdir() if f.is_dir()]
        elif content.lower() == "file":
            items = [f for f in p.iterdir() if f.is_file()]
        else:
            items = [f for f in p.iterdir()]

        # Filter by extension if specified (only applies to files)
        if match_extension:
            items = [f for f in items if f.suffix.lstrip(".").lower() == match_extension.lower()]

        return sorted([f.name for f in items])

    except Exception as e:
        print(f"Error: {e}")
        return False
