def init(test_name: str, test_case: str) -> None:
    """
    Create test folder and inject TOML blocks into config.ext.

    Args:
        test_name: name of the test task to be created
        test_case: name of the test case to be used for this test task, should be in TC_DIR

    Returns:
        None, creates a new test task folder with config.ext file in TEST_DIR

    Examples:
        >>> init("PMD-1234", "PMD-0000")
        This will create a new task named PMD-1234 in TEST_DIR, with a config.cmconf
    """
    tc_md = tui.add_extension_if_missing(test_case, ext)
    tc_path = paths.TC_DIR / tc_md

    config_template = f"config.{recipe_ext}.j2"

    init_data = {
        "task_name": test_name,
        "tc_dir": tc_md,
        "se_dir": paths.SE_DIR_REL.as_posix(),
    }
    tui.log_message(
        "DEBUG",
        f"Creating {test_name} with {tc_md} from config template: {config_template}",
        show_datetime=False,
    )
    if not tc_path.exists():
        tui.log_message("ERROR", f"Test case not found: {tc_path}. Exiting...", show_datetime=False)
        exit()

    out_dir = paths.TASKS_DIR / test_name
    out_dir.mkdir(parents=True, exist_ok=True)

    toml_blocks = extract_blocks(tc_path.read_text(), "TOML")
    config_template = jinja_env.get_template(config_template)

    tui.log_message(
        "DEBUG",
        f"initializing with\n{ren.render_dict_as_table(data=init_data, format="plaintext")}",
        show_datetime=False,
    )

    config_rendered = config_template.render(init=init_data)
    config_out = inject(config_rendered, toml_blocks)

    create_temp_toml(test_name, test_case)

    (out_dir / f"config.{recipe_ext}").write_text(config_out)
    tui.log_message(
        "SUCCESS",
        f"test task {test_name} created from test case {test_case} at {out_dir / f'config.{recipe_ext}'}",
        show_datetime=False,
    )

def fetch(test_name: str) -> None:
    """
    Extract INSTRUMENT_DEFINITION blocks -> instrument.toml.

    Args:
        test_name: name of the test task to fetch instrument definitions for, should be in TEST_DIR

    Returns:
        None, creates/overwrites instrument.toml in the test task folder with the extracted INSTRUMENT_DEFINITION blocks
    """
    out_dir = paths.TASKS_DIR / test_name
    config = load_toml(out_dir / f"config.{recipe_ext}")
    parts = []

    tui.log_message("DEBUG", f"loading config.{recipe_ext} from {out_dir}", show_datetime=False)

    for role, se_name in config.get("instrument", {}).items():
        if not se_name:
            continue
        blocks = extract_blocks(read_text(se_name), "TOML")
        if "INSTRUMENT_DEFINITION" in blocks:
            parts.append(blocks["INSTRUMENT_DEFINITION"])
            # vvv only for debugging
            data = tomllib.loads(blocks["INSTRUMENT_DEFINITION"])
            k = next(iter(data), None)
            s = data[k]
            tui.log_message(
                "DEBUG",
                f'Found "{k}"\n{ren.render_dict_as_table(data=s, format="plaintext")}',
                show_datetime=False,
            )

    if not parts:
        tui.log_message(
            "ERROR",
            f"Fetched config.{recipe_ext}, No instrument definitions found",
            show_datetime=False,
        )
        return

    (out_dir / "instrument.toml").write_text("\n\n".join(parts) + "\n")
    tui.log_message(
        "SUCCESS", f"Fetch OK. {out_dir / 'instrument.toml'} is created", show_datetime=False
    )

def get_steps(test_name: str, output: str = "str") -> str | list:
    """
    Extract TEST_STEPS block from test case and SEs, then process it with hr_test_logic to get the final test steps.

    Args:
        test_name: name of the test task to get steps for, should be in TEST_DIR
        output: "str" or "list", determines the output format, default is "str"

    Returns:
        If output is "str", returns a string with each step separated by a newline.
        If output is "list", returns a list of steps.
    """
    out_dir = paths.TASKS_DIR / test_name
    config = load_toml(out_dir / f"config.{recipe_ext}")

    robot_blocks = extract_blocks(
        (paths.TC_DIR / f"{config['env']['test_case']}").read_text(), "ROBOT"
    )
    res = hr_test_logic(robot_blocks["TEST_STEPS"])
    if output == "list":
        return res
    else:
        a = ""

        for line in res:
            a += line + "\n"
        return a

def read_text(se_name, module_separator="__"):
    """
    Read the content of a structural element file. If the name contains module separator, treat it as a reference to another module and fetch the content via API request.

    Args:
        se_name: name of the structural element file to read
        module_separator: separator used to identify references to other modules

    Returns:
        The content of the structural element file.
    """
    if module_separator in se_name:  # take it from another module
        inst = se_name.split(module_separator)
        tui.log_message(
            "INFO", f"Detected {inst[1]} from an external module: {inst[0]}", show_datetime=False
        )
        text = run_api_request(dotted_input=se_name, module_saparator="__")
    else:
        se_path = paths.SE_DIR / f"{se_name}.{ext}"
        if not se_path.exists():
            tui.log_message("ERROR", f"{se_path} doesn't exist", show_datetime=False)
            raise FileNotFoundError
        text = se_path.read_text()
    return text

def build(test_name: str) -> None:
    """
    Copy base template, then inject all ROBOT blocks from test case and SEs.
    Result: .robot.j2 with all blocks injected, Jinja2 variables intact.

    Args:
        test_name: name of the test task to build, should be in TEST_DIR

    Returns:
        None, creates/overwrites .robot.j2 file in the test task folder
    """
    out_dir = paths.TASKS_DIR / test_name
    config = load_toml(out_dir / f"config.{recipe_ext}")
    instrument_raw = load_toml(out_dir / "instrument.toml")

    tui.log_message("DEBUG", f"Building template of {out_dir}", show_datetime=False)

    base_j2 = (paths.TEMPLATES_DIR / "test_case.robot.j2").read_text()
    robot_blocks = extract_blocks(
        (paths.TC_DIR / f"{config['env']['test_case']}").read_text(), "ROBOT"
    )

    for se_name in instrument_raw:
        for name, content in extract_blocks(read_text(se_name), "ROBOT").items():
            if name in robot_blocks:
                robot_blocks[name] += "\n" + content
            else:
                robot_blocks[name] = content

    tui.log_message("DEBUG", get_steps(test_name, output="str"), show_datetime=False)
    assembled_j2 = inject(base_j2, robot_blocks)
    tui.log_message("DEBUG", "File is injected without error", show_datetime=False)
    output_path = out_dir / f"{test_name}.robot.j2"
    output_path.write_text(assembled_j2)
    tui.log_message(
        "SUCCESS", f"{test_name} template is created at {output_path}", show_datetime=False
    )

def render(test_name: str) -> None:
    """
    Read .robot.j2 and apply Jinja2 context from config.ext + instrument.toml

    Args:
        test_name: name of the test task to render, should be in TEST_DIR

    Returns:
        None, creates/overwrites .robot file in the test task folder
    """
    out_dir = paths.TASKS_DIR / test_name
    config = load_toml(out_dir / f"config.{recipe_ext}")
    instrument_raw = load_toml(out_dir / "instrument.toml")
    temp = out_dir / csv_parse_toml
    if temp.exists():
        dynamic_param = load_toml(temp)
        tui.log_message("DEBUG", f"Found {temp}", show_datetime=False)
    else:
        tui.log_message(
            "WARNING",
            f"{temp} doesn't exist, make sure that this test doesn't need it",
            show_datetime=False,
        )  # warn
        dynamic_param = {"parameter": {"dynamic": {"DEFAULT": {}}}}

    instrument_ctx = {
        role: instrument_raw[se_name]["name"]
        for role, se_name in config.get("instrument", {}).items()
        if se_name and se_name in instrument_raw
    }

    assembled_j2 = (out_dir / f"{test_name}.robot.j2").read_text()
    context = {
        "info": config.get("info", {}),
        "env": config.get("env", {}),
        "parameter": {**config.get("parameter", {}), **dynamic_param.get("parameter", {})},
        "instrument": instrument_ctx,
        **instrument_raw,
    }

    tui.log_message("DEBUG", f"Resolving {test_name} with {context}", show_datetime=False)
    rendered = jinja_env.from_string(assembled_j2).render(**context)
    tui.log_message("DEBUG", f"{test_name} resolved without error", show_datetime=False)
    output_path = out_dir / f"{test_name}.robot"
    output_path.write_text(rendered)
    tui.log_message(
        "SUCCESS", f"{test_name} test file is rendered at {output_path}", show_datetime=False
    )

def clean(test_name: str) -> None:
    """
    Remove built and rendered files.

    Args:
        test_name: name of the test task to clean, should be in TEST_DIR

    Returns:
        None, removes files from the test task folder
    """
    cur_test_dir = paths.TASKS_DIR / test_name
    built_file = cur_test_dir / f"{test_name}.robot.j2"
    rendered_file = cur_test_dir / f"{test_name}.robot"

    for file in [built_file, rendered_file]:
        if not file.exists():
            tui.log_message(
                "WARNING", f"Nothing to do here {file}, does not exist", show_datetime=False
            )
        else:
            os.remove(file)
            tui.log_message("SUCCESS", f"{file} is deleted", show_datetime=False)

def load_config(data: dict) -> None:
    """
    Recieved dict from API request, then create test task with the data and build the template

    Args:
        data: dict with keys "test_task", "test_case", "structural_elements" (list of SE names)

    Returns:
        None, creates a new test task folder with config.ext file in TEST_DIR, then builds the template

    Examples:
        >>> data ={
        >>> "test_case": "PMD-0000",
        >>> "test_task": "BJEQ-1100",
        >>> "structural_elements": [
        >>>     {"id": "BJEQ-4444", "title": "AutoTestLib__m4m_tcp"},
        >>>     {"id": "BJEQ-5555", "title": "AutoTestLib__mt786"},
        >>> ],
        >>> "parameters": {"WireConfiguration": "3P4W3CT"},
        >>> }
        >>> load_config(data)
        This will create a new test task named BJEQ-1100 with PMD-0000 as the test case, then inject the SE references into config.cmconf based on the SE titles and build the template.
        After that it will fetch the instrument definitions from the SEs and create instrument.toml in the test task folder.
    """
    tui.log_message(
        "DEBUG",
        f"Loading config from API request with data\n{ren.render_dict_as_table(data, format='plaintext')}",
        show_datetime=False,
    )

    init(data["test_task"], data["test_case"])
    config_path = paths.TASKS_DIR / data["test_task"] / f"config.{recipe_ext}"
    config = load_toml(config_path)

    # Load se_lib.toml for reference when injecting SEs, so that we can automatically set the instrument fields based on the SE titles
    se_ref_lib = load_main_config("se_lib.toml")

    # Inject SE references into config.toml
    for se_name in data["structural_elements"]:
        for role, _ in config.get("instrument", {}).items():
            if role in find_item_path(se_ref_lib, se_name["title"]):
                tui.log_message(
                    "INFO",
                    f"Setting field {role} to {se_name['title']} based on se_lib.toml reference",
                    show_datetime=False,
                )
                config["instrument"][role] = se_name["title"]

    #
    with open(config_path, "wb") as f:
        tomli_w.dump(config, f)
        tui.log_message(
            "SUCCESS",
            f"config.toml is updated with structural element references via an API request, now building instrument.toml",
            show_datetime=False,
        )
    fetch(data["test_task"])

def create_temp_toml(test_name: str, test_case: str) -> None:
    """
    Create a temporary TOML file from a CSV file.

    Args:
        test_name: name of the test task, should be in TEST_DIR
        test_case: name of the test case, should be in TC_DIR

    Returns:
        None, creates a new TOML file in the test task folder
    """
    cur_test_dir = paths.TASKS_DIR / test_name
    filename = csv_parse_toml
    tc_csv = tui.add_extension_if_missing(test_case, "csv")
    csv_path = paths.TC_DIR / tc_csv

    if not csv_path.exists():
        tui.log_message(
            "WARNING",
            f"{tc_csv} doesn't exist, make sure that {test_case} doesn't have any dynamic parameters",
            show_datetime=False,
        )
    else:
        tui.log_message(
            "DEBUG",
            f"Generating {filename} from {csv_path} at {cur_test_dir}\n{ren.render_csv_table(csv_path, format="plaintext")[0]}",
            show_datetime=False,
        )
        DynVarMat(csv_path).export_toml(cur_test_dir, fname=filename)
        tui.log_message("SUCCESS", f"{filename} is generated ag {cur_test_dir} from {tc_csv}")

def generate_doc(test_name: str):
    """
    Generate README.md for the test task by extracting info from config.toml and instrument.toml, then applying them to the README template.

    Args:
        test_name: name of the test task, should be in TEST_DIR

    Returns:
        str: README.md file, also saves the README.md file in the test task folder
    """
    cur_test_dir = paths.TASKS_DIR / test_name
    config_file = cur_test_dir / f"config.{recipe_ext}"
    instrument_file = cur_test_dir / "instrument.toml"

    config = load_toml(config_file)
    csv_file = paths.TC_DIR / f"{config["env"]["test_case"]}.csv"
    if csv_file.exists():
        csv_path = csv_file
    else:
        csv_path = ""

    readme = generate_readme(
        toml_files=[config_file, instrument_file],
        csv_file=csv_path,
        output_file=cur_test_dir / CONFIG["template"]["readme"],
        title_path="info.test_case.name",
        instrument_section="instrument",
    )
    return readme
