def load_main_config(config_file):
    """Load main configuration from TOML file with fallback to system default.

    Attempts to load user-specific config from virtual environment parent directory,
    falling back to system config if not found.

    Args:
        config_file (str): Name of the configuration file to load.

    Returns:
        dict: Parsed TOML configuration dictionary.

    Raises:
        FileNotFoundError: If config file cannot be found in any location.
    """
    # try defualt location for user that is using venv
    venv_path = Path(sys.prefix)
    config_path = venv_path.resolve().parent / "config" / config_file

    if config_path.exists():
        tui.log_message("DEBUG", f"Found user config file {config_file} at {config_path}")
    else:
        tui.log_message(
            "DEBUG", f"No user config file {config_file} found, using system's {config_file}"
        )
        config_path = Path(__file__).resolve().parent.parent / config_file

    if not config_path.exists():
        raise FileNotFoundError(f"Config file not found: {config_path}")

    with open(config_path, "rb") as f:
        config = tomllib.load(f)

    return config

def extract_blocks(text: str, tag: str) -> dict[str, str]:
    """Extract content from markdown code blocks with markers.

    Parses markdown text and extracts named blocks delimited by marker comments.

    Args:
        text (str): Markdown text containing code blocks.
        tag (str): Tag name to identify blocks (e.g., "PYTHON", "ROBOT").

    Returns:
        dict[str, str]: Dictionary mapping block names to their content.
    """
    pattern = re.compile(
        rf"### {tag}-BLOCK-START: (\w+)\n(.*?)\n### {tag}-BLOCK-END: \1",
        re.DOTALL,
    )
    return {name: content.rstrip() for name, content in pattern.findall(text)}

def inject(template: str, blocks: dict[str, str]) -> str:
    """Replace content between INJECTION-POINT markers with block content.

    Substitutes placeholder regions in a template with extracted block content.

    Args:
        template (str): Template text containing injection point markers.
        blocks (dict[str, str]): Dictionary of block names and content to inject.

    Returns:
        str: Template with injected content.
    """
    for name, content in blocks.items():
        pattern = re.compile(
            rf"(### INJECTION-POINT-START: {name}\n)(.*?)(### INJECTION-POINT-END: {name})",
            re.DOTALL,
        )
        replacement = rf"\g<1>{content}\n\g<3>"
        template = pattern.sub(replacement, template)
    return template

def hr_test_logic(template_string):
    """Convert Robot Framework test template to human-readable test logic.

    Parses Robot Framework syntax and transforms it into natural language
    descriptions of test steps, removing code-specific syntax and path prefixes.

    Args:
        template_string (str): Robot Framework test template as multi-line string.

    Returns:
        list[str]: List of human-readable test step descriptions.
    """

    def deep_clean(text):
        """Nukes all code-specific syntax and path prefixes."""
        # 1. Remove Jinja braces {{ }}
        text = re.sub(r"\{\{ ?| ?\}\}", "", text)
        # 2. Remove common path prefixes
        text = re.sub(r"(instrument|parameter|static)\.", "", text)
        # 3. Strip Robot syntax ${} @{} &{}
        text = re.sub(r"[\$\@\&]\{|\}", "", text)
        # 4. Final dot-strip for remaining namespaces (if not a number/version)
        if "." in text and not re.match(r"^\d", text):
            text = text.split(".")[-1]
        return text.strip()

    def get_smart_preposition(action, has_target=False):
        """Selects the right connector based on the verb and context."""
        action = action.lower()
        if any(word in action for word in ["sleep", "wait"]):
            return "for"
        if has_target:
            if any(word in action for word in ["get", "read", "fetch", "measure"]):
                return "from"
            if any(word in action for word in ["set", "write", "apply", "send", "configure"]):
                return "to"
            return "on"
        return ""

    manifest = []
    sk = False
    for line in template_string.splitlines():
        line = line.strip()
        if not line or line.startswith(("***", "#")):
            continue

        if line.startswith(("FOR", "WHILE", "IF")):
            sk = True

        # Step 1: Inline Loop Simplifier (strips loop logic, keeps the iterable)
        line = re.sub(
            r"\{%\s*for\s+\w+\s+in\s+([^%]+)%\}.*?\{%\s*endfor\s*%\}",
            lambda m: f"    {m.group(1).strip()}",
            line,
        )

        # Step 2: Split by Robot separators (2+ spaces or tabs)
        cells = re.split(r"\s{2,}|\t", line)
        if not cells[0]:
            continue

        # Step 3: Identify Assignment
        if cells[0].endswith("="):
            target_var = deep_clean(cells[0].replace("=", ""))
            raw_action = cells[1]
            args = cells[2:]
        else:
            target_var = None
            raw_action = cells[0]
            args = cells[1:]

        # Step 4: Process Action and Prep
        has_target = "." in raw_action
        if has_target:
            namespace, method = raw_action.rsplit(".", 1)
            target_on = deep_clean(namespace)
            action_name = method.strip()
            prep = get_smart_preposition(action_name, has_target=True)
            action_str = f"[{action_name}] {prep} {target_on}"
        else:
            action_name = deep_clean(raw_action)
            prep = get_smart_preposition(action_name, has_target=False)
            action_str = f"[{action_name}] {prep}".strip()

        # Step 5: Linguistic Assembly (Avoiding "for using")
        clean_args = [deep_clean(a) for a in args]
        if prep in ["for", "from", "to"] and clean_args:
            arg_str = f" {', '.join(clean_args)}"
        elif clean_args:
            arg_str = f" using {', '.join(clean_args)}"
        else:
            arg_str = ""

        store_str = f" -> store in {target_var}" if target_var else ""

        # Merge and clean whitespace
        final_line = f"{action_str}{arg_str}{store_str}"
        manifest.append(re.sub(r"\s+", " ", final_line).strip())

    return manifest

def run_api_request(dotted_input: str, module_saparator="__"):
    """
    Parse "ModuleName.method_name" and dynamically run:
        ModuleName.api.request("method_name")

    Args:
        dotted_input (str): Input string in the format "ModuleName.method_name"
                            e.g. "AutoTestLib__m4m_tcp"

    Returns:
        Result of ModuleName.api.request_se("method_name")
    """
    # Split into module name and request name
    parts = dotted_input.split(module_saparator, 1)  # Split only on the first dot
    if len(parts) != 2:
        raise ValueError(
            f"Invalid input format: '{dotted_input}'. Expected 'ModuleName.method_name'"
        )

    module_name, request_name = parts

    # Dynamically import the module
    try:
        module = importlib.import_module(module_name)
    except ModuleNotFoundError:
        raise ModuleNotFoundError(f"Module '{module_name}' could not be found.")

    # Call module.api.request(request_name)
    try:
        result = module.api.request_se(request_name)
    except AttributeError as e:
        raise AttributeError(
            f"'{module_name}' does not have the expected 'api.request' interface: {e}"
        )

    return result

def find_item_path(config_dict, target_item):
    """
    Find the path to an item in the nested config dictionary.

    Args:
        config_dict: The nested dictionary to search
        target_item: The item to find

    Returns:
        List of keys representing the path, or None if not found
    """

    def search(d, path=[]):
        for key, value in d.items():
            if key == "items" and isinstance(value, list):
                if target_item in value:
                    return path
            elif isinstance(value, dict):
                result = search(value, path + [key])
                if result is not None:
                    return result
        return None

    return search(config_dict)
