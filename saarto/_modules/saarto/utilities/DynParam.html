class DynVarMat:
    """Class to handle dynamic parameter matrix from CSV file, and export to TOML format for test configuration"""

    def __init__(self, csv_file_path):
        """
        Initialize the dynamic parameter matrix by reading from a CSV file.

        Arg:
            csv_file_path (str or Path): Path to the CSV file containing the dynamic parameters. The first row should contain the keys, and the first two columns should be description and name respectively.

        Returns:
            None, initializes the data matrix and prepares the data for export.
        """
        self.data_matrix = []
        with open(csv_file_path, newline="") as csvfile:
            csvreader = csv.reader(csvfile, delimiter=";")
            for row in csvreader:
                self.data_matrix.append(DynVar(row))
        # obtain keys of a dict also remove the first row
        self.keys = [k for k in self.data_matrix.pop(0).data if k != None]
        self.names = []
        for i in self.data_matrix:
            self.names.append(i.name)
        # rotate the matrix
        temp_mat = [x.data for x in self.data_matrix]
        self.values = [list(row) for row in zip(*temp_mat)][0 : self.__len__()[1]]
        # build them into a dictionary
        self.data = dict(zip(self.keys, self.values))

    def export_toml(self, test_folder_path: str | Path, fname="temp.toml"):
        """
        Export the dynamic parameter matrix to a TOML file in the specified test folder.
        Args:
            test_folder_path (str or Path): The path to the test folder where the TOML file will be saved.
            fname (str): The name of the TOML file to be created (default is "temp.toml").
        Returns:
            None, creates a TOML file with the dynamic parameters in the specified location.
        """

        def convert_none(val):
            """Recursively convert None values to "None" string"""
            if val is None:
                return "None"
            elif isinstance(val, list):
                return [convert_none(item) for item in val]
            else:
                return val

        # Build nested dict
        toml_dict = {"parameter": {"dynamic": {}}}
        for section, values in self.data.items():
            toml_dict["parameter"]["dynamic"][section] = {
                key: convert_none(val) for key, val in zip(self.names, values)
            }  # **** parse None to "None" since toml doesn't support

        # Add file name
        toml_dict["test"] = test_folder_path.name.split(".")[0]

        # Write directly to file
        try:
            with open(test_folder_path / fname, "wb") as f:  # must be binary mode for tomli-w
                tomli_w.dump(toml_dict, f)
            print("TOML file saved successfully as config.toml")
        except (OSError, TypeError) as e:
            print(f"Error saving TOML file: {e}")

class DynVar:
    """Class to represent a single dynamic variable with its name, description, and data values"""

    def __init__(self, list_in=[]):
        """
        Initialize a dynamic variable from a list of values, where the first two elements are description and name, and the rest are data values.

        Args:
            list_in (list): A list containing the description, name, and data values of the dynamic variable. The first element is the description, the second is the name, and the rest are data values.

        Returns:
            None, initializes the dynamic variable with the provided information.
        """
        list_in = [None, None] if not list_in else list_in
        self.description = list_in.pop(0)
        self.name = list_in.pop(0)
        self.data = []
        for raw in list_in:
            self.data.append(DynVar.smart_type_parser(raw))

    def __repr__(self):
        return f"variable: {self.name}, data: {self.data}, description: {self.description}"

    def __str__(self):
        return self.__repr__()

    @staticmethod
    def smart_type_parser(item: str):
        """
        Parse string value to appropriate Python type:
        - "3P3W2CT"          → str
        - "True" / "False"   → bool
        - "123" / "1.23"     → int / float
        - "213/123/None"     → list of int/float/None

        Args:
            item (str): The string value to be parsed.
        Returns:
            The parsed value in the appropriate Python type (str, bool, int, float, list).
        """

        if not isinstance(item, str) or item.strip() == "":
            return None

        item = item.strip()

        # check for boolean
        if item.lower() == "true":
            return True
        if item.lower() == "false":
            return False

        # check for list (slash-separated)
        if "/" in item:
            parts = item.split("/")
            return [DynVar.smart_type_parser(p.strip()) for p in parts]

        # check for None
        if item.lower() == "none":
            return None

        # check for int
        try:
            if "." not in item:
                return int(item)
        except ValueError:
            pass

        # check for float
        try:
            return float(item)
        except ValueError:
            pass

        # fallback : string
        return str(item).replace('"', "").replace("'", "")
