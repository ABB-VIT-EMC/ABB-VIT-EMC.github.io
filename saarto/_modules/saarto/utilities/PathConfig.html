class PathConfig:
    """Load and manage project paths from TOML configuration."""

    _instance = None
    _paths = {}

    def __new__(cls, config_file="config.toml"):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance._set_paths(config_file)
        return cls._instance

    def _set_paths(self, config_file):
        config = load_main_config(config_file)

        paths = config["paths"]
        saarto_internal = Path(__file__).resolve().parent.parent

        if paths["use_env"]:
            base_dir = Path(sys.prefix).parent.resolve()
        else:
            base_dir = saarto_internal / paths["saarto_root"]

        # Resolve all paths relative to base directory
        self._paths = {
            "HERE": (base_dir / paths["here"]).resolve(),
            "SAARTO_ROOT": (base_dir).resolve(),
            "TEMPLATES_DIR": (saarto_internal / paths["templates_dir"]).resolve(),
            "TC_DIR_REL": Path(paths["tc_dir_rel"]),
            "TC_DIR": (base_dir / paths["tc_dir_rel"]).resolve(),
            "SE_DIR_REL": Path(paths["se_dir_rel"]),
            "SE_DIR": (base_dir / paths["se_dir_rel"]).resolve(),
            "TASKS_DIR_REL": Path(paths["tasks_dir_rel"]),
            "TASKS_DIR": (base_dir / paths["tasks_dir_rel"]).resolve(),
        }

    def __getattr__(self, name):
        """Access paths as attributes."""
        if name in self._paths:
            return self._paths[name]
        raise AttributeError(f"Path '{name}' not found in config")

    def get(self, name, default=None):
        """Get path with fallback default."""
        return self._paths.get(name, default)



    def __repr__(self):
        """Display all available paths."""
        return "\n".join(f"{k}: {v}" for k, v in self._paths.items())
