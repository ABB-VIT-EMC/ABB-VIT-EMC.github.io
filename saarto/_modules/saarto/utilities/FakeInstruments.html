    def get_data(self, reg: str|list):
        if isinstance(reg, str):
            value = self.registers.get(reg, None)
            logger.info(f"Read register 0x{reg:04X}: {value}")
        elif isinstance(reg, list):
            value = []
            for r in reg:
                value.append(self.registers.get(r, None))
            logger.info(f"Read a list of registers: {value}")
        
        return value
