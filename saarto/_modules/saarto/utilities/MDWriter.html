def snake_to_title_case(snake_str):
    return ' '.join(word.capitalize() for word in snake_str.split('_'))

def render_section(title: str, data, level: int = 2) -> str:
    """
    Recursively render any TOML section as readable markdown.
    - Flat dict   -> markdown table
    - Nested dict -> subsections (recursive)
    - List        -> bullet points
    - Scalar      -> inline code value
    """
    heading = "#" * level
    lines = [f"{heading} {snake_to_title_case(title)}\n\n"]

    if isinstance(data, dict):
        scalars     = {k: v for k, v in data.items() if not isinstance(v, (dict, list))}
        nested_dicts = {k: v for k, v in data.items() if isinstance(v, dict)}
        nested_lists = {k: v for k, v in data.items() if isinstance(v, list)}

        if scalars:
            lines.append(render_dict_as_table(scalars))

        for k, v in nested_lists.items():
            lines.append(f"**{k}:**\n\n")
            lines.extend(f"- {item}\n" for item in v) if v else lines.append("*empty*\n")
            lines.append("\n")

        for k, v in nested_dicts.items():
            lines.append(render_section(k, v, level + 1))

    elif isinstance(data, list):
        lines.extend(f"- {item}\n" for item in data) if data else lines.append("*empty*\n")
        lines.append("\n")

    else:
        lines.append(f"{render_value(data)}\n\n")

    return "".join(lines)

def render_instrument_section(instruments: dict, lookup: dict, level: int = 2) -> str:
    """
    Render instrument section with cross-reference lookup from other TOML files.
    Each instrument alias maps to a key searched in the lookup dict.
    """
    heading = "#" * level
    lines = [f"{heading} Instrument\n\n"]

    for alias, key in instruments.items():
        lines.append(f"{'#' * (level + 1)} {alias}\n\n")

        if key in lookup:
            lines.append(render_dict_as_table(lookup[key], "Configuration", "Value"))
        else:
            lines.append("*No configuration found in provided instrument files.*\n\n")

    return "".join(lines)

def resolve_path(data: dict, dot_path: str) -> dict:
    """Resolve a dot-separated key path from a nested dict (e.g. 'info.test_case.name')."""
    result = data
    for k in dot_path.split("."):
        if isinstance(result, dict):
            result = result.get(k)
        else:
            return None
    return result

def generate_readme(
    toml_files: list,
    csv_file: str | None,
    output_file: str = "README.md",
    title_path = None,
    instrument_section = "instrument",
    skip_sections = None,
) -> str:
    """
    Generate a GitHub README from TOML configs and a CSV test variation file.

    Args:
        toml_files        : List of TOML files.
                            First  = main config (drives the README structure).
                            Others = lookup configs (e.g. instrument definitions).
        csv_file          : Semicolon-delimited CSV with test variations.
        output_file       : Output path for the generated README (default: "README.md").
        title_path        : Dot-separated path to the title field in main TOML,
                            e.g. "info.test_case.name". Falls back to the file name.
        instrument_section: Key in main TOML holding instrument alias->class mappings.
                            Set to None to skip instrument cross-referencing.
        skip_sections     : Top-level TOML keys to exclude from the README.

    Returns:
        Generated README content as a string.

    Example:
        generate_readme(
            toml_files=["main_config.toml", "instruments_config.toml"],
            csv_file="test_variations.csv",
            output_file="README.md",
            title_path="info.test_case.name",
        )
    """
    skip_sections = skip_sections or []

    # Load configs
    main_config = load_toml(toml_files[0])
    lookup = {}
    for f in toml_files[1:]:
        lookup.update(load_toml(f))

    # Resolve title
    title = resolve_path(main_config, title_path) if title_path else None
    if not title or isinstance(title, dict):
        title = output_file.replace(".md", "")

    toc_entries    = []
    sections_content = []

    # Render each top-level section dynamically
    for section_name, section_data in main_config.items():
        if section_name in skip_sections:
            continue

        if section_name == instrument_section and isinstance(section_data, dict):
            content = render_instrument_section(section_data, lookup, level=2)
        else:
            content = render_section(section_name, section_data, level=2)

        anchor = section_name.lower().replace(" ", "-")
        toc_entries.append(f"- [{snake_to_title_case(section_name)}](#{anchor})\n")
        sections_content.append(content)

    toc_entries.append("- [Iterations](#test-variations)\n")

    # Assemble README
    readme = []
    readme.append(f"# {title}\n\n")
    readme.append(f"> *this readme is auto generated*\n\n")
    readme.append(" Table of Contents\n\n")
    readme.extend(toc_entries)
    readme.append("\n")
    readme.extend(sections_content)
    if csv_file:
        readme.append("### Parameter Table\n\n")
        csv_table, num_cases = render_csv_table(csv_file)
        readme.append(f"*Total test cases: {num_cases}*\n\n")
        readme.append(csv_table)
    readme.append("\n---\n\n")
    readme.append(f"> *Last generated: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}*\n")

    content = "".join(readme)

    with open(output_file, "w") as f:
        f.write(content)

    tui.log_message("SUCCESS", f"README generated: {output_file}")
    return content
