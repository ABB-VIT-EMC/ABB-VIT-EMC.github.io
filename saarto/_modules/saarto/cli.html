def generate_commands_markdown():
    """
    Generate commands table.
    """
    # Markdown table for README
    markdown = "| Command | Description | Arguments |\n"
    markdown += "|---------|-------------|----------|\n"

    for cmd_name, cmd_config in COMMANDS.items():
        args = ", ".join([f"`{arg[0]}`" for arg in cmd_config["args"]])
        markdown += f"| `{cmd_name}` | {cmd_config['help']} | {args} |\n"

    return markdown

def inject_commands_to_readme():
    """Inject generated commands table into README.md"""
    readme_path = Path(__file__).parent.parent.parent / "README.md"
    if not readme_path.exists():
        print("README.md not found")
        return

    content = readme_path.read_text()

    # Replace content between markers
    pattern = r"(<!-- COMMANDS_START -->\s*\n)(.*?)(<!-- COMMANDS_END -->)"
    replacement = rf"\1{generate_commands_markdown()}\n\3"

    new_content = re.sub(pattern, replacement, content, flags=re.DOTALL)
    readme_path.write_text(new_content)

def main() -> None:
    inject_commands_to_readme()
    parser = argparse.ArgumentParser(
        description="saarto --- System for Automated, Anti-Redundancy Test Orchestrator",
        formatter_class=argparse.RawDescriptionHelpFormatter,
        epilog="",
    )
    subparsers = parser.add_subparsers(dest="command", help="Available commands")

    # Create parsers dynamically
    for cmd_name, cmd_config in COMMANDS.items():
        subparser_i = subparsers.add_parser(cmd_name, help=cmd_config["help"])

        # Add arguments
        for arg in cmd_config["args"]:
            subparser_i.add_argument(arg[0], help=arg[1])

        # Auto-derive function from key name
        subparser_i.set_defaults(func=globals()[f"cmd_{cmd_name}"])

    args = parser.parse_args()

    if not args.command:
        parser.print_help()
        sys.exit(0)

    try:
        args.func(args)
    except Exception as exc:
        tui.log_message("ERROR", f"{exc}")
        raise
