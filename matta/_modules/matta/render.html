def render_value(value) -> str:
    """Render a scalar value as markdown inline code.

    Args:
        value: The value to render. Can be None, empty string, or any scalar.

    Returns:
        str: Markdown formatted value (backticks) or "*N/A*" if None or empty.
    """
    if value is None or value == "":
        return "*N/A*"
    return f"`{value}`"

def render_dict_as_table(
    data: dict, key_col: str = "Key", val_col: str = "Value", format: str = "markdown"
) -> str:
    """Render a flat dictionary as a formatted table.

    Args:
        data (dict): Flat dictionary to render. Nested dicts and lists are skipped.
        key_col (str): Header name for the key column. Defaults to "Key".
        val_col (str): Header name for the value column. Defaults to "Value".
        format (str): Output format, either "markdown" or "plaintext".
            Defaults to "markdown".

    Returns:
        str: Table formatted as markdown or plaintext.
    """
    if format == "plaintext":
        # Terminal/plaintext format
        key_width = max(len(str(k)) for k in data.keys()) if data else len(key_col)
        key_width = max(key_width, len(key_col))
        val_width = (
            max(len(str(render_value(v))) for v in data.values() if not isinstance(v, (dict, list)))
            if data
            else len(val_col)
        )
        val_width = max(val_width, len(val_col))

        lines = []
        lines.append(f"{key_col:<{key_width}}  {val_col:<{val_width}}\n")
        lines.append(f"{chr(45)*key_width}  {chr(45)*val_width}\n")
        for k, v in data.items():
            if not isinstance(v, (dict, list)):
                lines.append(f"{str(k):<{key_width}}  {render_value(v):<{val_width}}\n")
        return "".join(lines)
    else:
        # Markdown format (original)
        lines = [
            f"| {key_col} | {val_col} |\n",
            f"|{chr(45)*(len(key_col)+2)}|{chr(45)*(len(val_col)+2)}|\n",
        ]
        for k, v in data.items():
            if not isinstance(v, (dict, list)):
                lines.append(f"| {k} | {render_value(v)} |\n")
        return "".join(lines) + "\n"

def render_csv_table(csv_file: str, format: str = "markdown") -> tuple[str, int]:
    """Render CSV test variations as a formatted table.

    Parses semicolon-delimited CSV with test case parameters and variations.

    CSV Convention:
        - Row 0 (header): col 0 = label, col 1 = ignored, col 2+ = test case names
        - Row 1+: col 0 = param name, col 1 = short label, col 2+ = values

    Args:
        csv_file (str): Path to the CSV file.
        format (str): Output format, either "markdown" or "plaintext".
            Defaults to "markdown".

    Returns:
        tuple: (table_string, num_test_cases) where table_string is the formatted
            table and num_test_cases is the count of test case columns.
    """
    with open(csv_file, "r", encoding="utf-8") as f:
        reader = csv.reader(f, delimiter=";")
        header = next(reader)
        test_case_names = [h for h in header[2:] if h.strip()]

        rows = []
        for row in reader:
            if len(row) >= 2:
                param_name = row[0]
                param_short = row[1]
                values = row[2 : len(test_case_names) + 2]
                rows.append((param_name, param_short, values))

    num_cases = len(test_case_names)

    if format == "plaintext":
        # Terminal/plaintext format
        col_widths = [len("Parameter"), len("Short")] + [len(name) for name in test_case_names]
        for param_name, param_short, values in rows:
            col_widths[0] = max(col_widths[0], len(param_name))
            col_widths[1] = max(col_widths[1], len(render_value(param_short)))
            for i, v in enumerate(values):
                col_widths[2 + i] = max(col_widths[2 + i], len(str(v)))

        lines = []
        header_row = f"{'Parameter':<{col_widths[0]}}  {'Short':<{col_widths[1]}}"
        header_row += "".join(
            f"  {name:<{col_widths[2 + i]}}" for i, name in enumerate(test_case_names)
        )
        lines.append(header_row + "\n")
        lines.append("-" * len(header_row) + "\n")

        for param_name, param_short, values in rows:
            row_str = f"{param_name:<{col_widths[0]}}  {render_value(param_short):<{col_widths[1]}}"
            row_str += "".join(f"  {v:<{col_widths[2 + i]}}" for i, v in enumerate(values))
            lines.append(row_str + "\n")

        return "".join(lines), num_cases
    else:
        # Markdown format (original)
        lines = []
        lines.append("| Parameter | Short | " + " | ".join(test_case_names) + " |\n")
        lines.append("|-----------|-------|" + "---------|" * num_cases + "\n")

        for param_name, param_short, values in rows:
            row_str = f"| {param_name} | {render_value(param_short)} |"
            row_str += "".join(f" {v} |" for v in values)
            lines.append(row_str + "\n")

        return "".join(lines), num_cases

def build_tree_from_path(path=".", max_depth=3, current_depth=0, ignore_dirs=None):
    """Build a tree structure dictionary from filesystem paths.

    Recursively traverses directories up to a maximum depth and creates a nested
    dictionary representation. Common development directories are ignored by default.

    Args:
        path (str): Directory path to scan. Defaults to current directory.
        max_depth (int): Maximum depth to traverse. Defaults to 3.
        current_depth (int): Current recursion depth. Internal use only.
        ignore_dirs (set): Directory names to ignore. Defaults to
            {'.git', '__pycache__', '.venv', 'node_modules', '.pytest_cache'}.

    Returns:
        dict: Nested dictionary where keys are folder names, values are subdirectories,
            and "_files" key contains list of files in that directory.
    """
    if ignore_dirs is None:
        ignore_dirs = {".git", "__pycache__", ".venv", "node_modules", ".pytest_cache"}

    try:
        p = Path(path)
        if not p.exists() or not p.is_dir():
            return {}

        if current_depth >= max_depth:
            return {}

        tree = {}

        # Get folders
        folders = list_directory(path, content="folder")
        if folders:
            for folder in folders:
                if folder not in ignore_dirs:
                    subfolder_path = str(Path(path) / folder)
                    tree[folder] = build_tree_from_path(
                        subfolder_path, max_depth, current_depth + 1, ignore_dirs
                    )

        # Get files
        files = list_directory(path, content="file")
        if files:
            if "_files" not in tree:
                tree["_files"] = []
            tree["_files"] = files

        return tree

    except Exception as e:
        print(f"Error building tree: {e}")
        return {}

        tree_data (dict): Pre-built tree dictionary. If None, builds from path.
        path (str): Directory path to scan if tree_data is None. Defaults to ".".
        max_depth (int): Maximum depth to traverse. Defaults to 3.

        tree_data = build_tree_from_path(path, max_depth)

def render_tree_dict(tree_dict, root_name="root"):
    """Render a manually created tree dictionary with ASCII art formatting.

    Args:
        tree_dict (dict): Dictionary representing tree structure.
        root_name (str): Name of the root node. Defaults to "root".

    Returns:
        str: ASCII art tree representation.
    """
    lines = []
    lines.append(f"{root_name}/")
    _render_tree_recursive(tree_dict, lines, "", True)

    return "\n".join(lines)
